/** Macro ucounty - Start Definition **/

%macro ucounty( ucountyv, varlbl, statev, countyv );

length
  &ucountyv $ 5
;

if &statev ~= "" and &countyv ~= "" then
  &ucountyv = &statev || &countyv;

label
  &ucountyv = "&varlbl"
;

%mend ucounty;

/** End Macro Definition **/
