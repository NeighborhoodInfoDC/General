%macro Block10_to_bpk( invar=geoblk2010, outvar=bridgepk, format=Y );

  length &outvar $ 1;
  
  &outvar = put( &invar, $bk1bpk. );
  
  label &outvar = "11th Street Bridge Park Target Area (2017)";
  
  %if %upcase( &format ) = Y %then %do;
    format &outvar $bpka.;
  %end;

%mend Block10_to_bpk;
