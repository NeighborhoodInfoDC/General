%macro Block00_to_regcnt( invar=geoblk2000, outvar=county, format=Y );

  length &outvar $ 1;
  
  &outvar = put( &invar, $bk0regcnt. );
  
  label &outvar = "Regional counties (2017)";
  
  %if %upcase( &format ) = Y %then %do;
    format &outvar $regcnta.;
  %end;

%mend Block00_to_regcnt;
