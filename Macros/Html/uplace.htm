/** Macro uplace - Start Definition **/

%macro uplace( uplacev, varlbl, statev, placev );

length
  &uplacev $ 7
;

if &statev ~= "" and &placev ~= "" then
  &uplacev = &statev || &placev;

label
  &uplacev = "&varlbl"
;

%mend uplace;

/** End Macro Definition **/
