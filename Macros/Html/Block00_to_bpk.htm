%macro Block00_to_bpk( invar=geoblk2000, outvar=bridgepk, format=Y );

  length &outvar $ 1;
  
  &outvar = put( &invar, $bk0bpk. );
  
  label &outvar = "11th Street Bridge Park Target Area (2017)";
  
  %if %upcase( &format ) = Y %then %do;
    format &outvar $bpka.;
  %end;

%mend Block00_to_bpk;
